function orbitalPeriod(arr) {
  var GM = 398600.4418;
  var earthRadius = 6367.4447;
  let arrRet = [];
  let currOP;
  for (let i in arr) {
    currOP = 2*Math.PI*Math.sqrt(Math.pow(earthRadius+arr[i].avgAlt,3)/GM);
    arrRet.push({"name":arr[i].name,
                "orbitalPeriod":Math.round(currOP)});
    console.log(Math.round(currOP));
  }
  return arrRet;
}


orbitalPeriod([{name : "sputnik", avgAlt : 35873.5553}]);
