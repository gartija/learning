function sumPrimes(num) {
  let primeSum = 0;
  for(let i = 2; i<=num; i++) {
    let isPrime = true;
    for(let j = i-1 ; j>1 ; j-- ) {
      if(i%j === 0) {
        
        isPrime = false; 
      }
    }
    if(isPrime) {
      primeSum +=i;
    }
  }
  console.log(primeSum);
  return primeSum;
}

sumPrimes(10);
