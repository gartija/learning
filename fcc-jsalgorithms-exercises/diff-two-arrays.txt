function diffArray(arr1, arr2) {
  var newArr = arr1.concat(arr2).sort((a,b) => a-b);
  newArr = newArr.filter((currVal) =>  {
      if(arr1.indexOf(currVal) == -1 || arr2.indexOf(currVal) == -1) {
        return true;
      }
      return false;
    }
  );
  console.log(newArr);
  return newArr;
}
